// ═══════════════════════════════════════════════════════════════════════
// قواعد أمان Firestore — انسخ المحتوى ده كله والصقه في:
// Firebase Console → Firestore Database → Rules → ثم اضغط "نشر / Publish"
// ═══════════════════════════════════════════════════════════════════════
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isSignedIn() { return request.auth != null; }
    function myUid() { return request.auth.uid; }
    function myRole() {
      return get(/databases/$(database)/documents/users/$(myUid())).data.role;
    }
    function isAdmin() { return isSignedIn() && myRole() == 'admin'; }

    // هل تم إعداد أول أدمن للنظام بالفعل؟ (يُستخدم في صفحة setup.html)
    function setupDone() {
      return exists(/databases/$(database)/documents/system/meta) &&
             get(/databases/$(database)/documents/system/meta).data.setupDone == true;
    }

    // مستند حالة النظام: قراءة عامة (تحتاجها صفحة تسجيل الدخول وصفحة الإعداد
    // قبل ما يكون فيه أي مستخدم مسجّل دخول)، وكتابة مسموحة مرة واحدة بس
    // (من false إلى true) عند إنشاء أول أدمن.
    match /system/meta {
      allow read: if true;
      allow create: if isSignedIn() && request.resource.data.setupDone == true;
      allow update: if isSignedIn() && resource.data.setupDone != true
                    && request.resource.data.setupDone == true;
      allow delete: if false;
    }

    // ملفات المستخدمين (الاسم، البريد، الدور: admin/employee)
    match /users/{uid} {
      allow read: if isSignedIn() && (myUid() == uid || isAdmin());
      // حالتان فقط مسموح فيهما إنشاء ملف مستخدم:
      // 1) الأدمن الحالي بينشئ حساب موظف جديد (من "متابعة الموظفين")
      // 2) صفحة setup.html بتنشئ أول حساب أدمن للنظام، وده بس لو لسه محدّش عمل ده (setupDone() == false)،
      //    وبشرط إن الحساب ده بينشئ نفسه (myUid() == uid)
      allow create: if isSignedIn() && (
                      (isAdmin() && request.resource.data.role == 'employee') ||
                      (myUid() == uid && request.resource.data.role == 'admin' && !setupDone())
                    );
      // كل مستخدم (أدمن أو موظف) يقدر يعدّل اسمه الظاهر بنفسه من "حسابي" — بدون المساس بالدور أو البريد
      allow update: if isAdmin() || (
                      isSignedIn() && myUid() == uid &&
                      request.resource.data.diff(resource.data).affectedKeys().hasOnly(['name'])
                    );
      allow delete: if isAdmin();
    }

    // المشاريع: الأدمن يدير كل شيء. الموظف يقرأ مشاريعه، يحدّث تقدمه هو فقط،
    // وكمان يقدر يُنشئ مشروعاً خاصاً به (يُسند لنفسه) ويعدّل/يحذف المشاريع التي أنشأها هو بنفسه.
    match /projects/{projectId} {
      allow read: if isAdmin() || (isSignedIn() && myUid() in resource.data.assignees);
      allow create: if isAdmin() || (
        isSignedIn() &&
        myUid() in request.resource.data.assignees &&
        request.resource.data.createdByUid == myUid()
      );
      allow update: if isAdmin() || (
        isSignedIn() && myUid() in resource.data.assignees &&
        (
          request.resource.data.diff(resource.data).affectedKeys().hasOnly(['progressMap']) ||
          (
            resource.data.createdByUid == myUid() &&
            request.resource.data.diff(resource.data).affectedKeys().hasOnly(['title','description'])
          )
        )
      );
      allow delete: if isAdmin() || (isSignedIn() && resource.data.createdByUid == myUid());
    }

    // الإنجازات: كل موظف يقرأ/يضيف إنجازاته فقط، الأدمن يقرأ كل شيء
    match /achievements/{achId} {
      allow read: if isAdmin() || (isSignedIn() && resource.data.uid == myUid());
      allow create: if isSignedIn() && request.resource.data.uid == myUid();
      allow update, delete: if isAdmin();
    }

    // الطلبات: الموظف يقدّم طلباته ويقرأها، الأدمن يقرأ الكل ويوافق/يرفض
    match /requests/{reqId} {
      allow read: if isAdmin() || (isSignedIn() && resource.data.uid == myUid());
      allow create: if isSignedIn() && request.resource.data.uid == myUid()
                    && request.resource.data.status == 'pending';
      allow update: if isAdmin();
      allow delete: if isAdmin();
    }

    // التقارير الأسبوعية: كل موظف يكتب تقريره الأسبوعي ويقرأ تقاريره فقط، الأدمن يقرأ تقارير الجميع
    match /weeklyReports/{repId} {
      allow read: if isAdmin() || (isSignedIn() && resource.data.uid == myUid());
      allow create: if isSignedIn() && request.resource.data.uid == myUid();
      allow update, delete: if isAdmin();
    }
  }
}
