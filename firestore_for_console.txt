rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isSignedIn() { return request.auth != null; }
    function myUid() { return request.auth.uid; }
    
    function myRole() {
      return get(/databases/$(database)/documents/users/$(myUid())).data.role;
    }
    
    function isAdmin() { 
      return isSignedIn() && myRole() == 'admin'; 
    }
    
    function isAnyAdmin() {
      return isSignedIn() && (myRole() == 'admin' || myRole() == 'assistant_admin');
    }

    function setupDone() {
      return exists(/databases/$(database)/documents/system/meta) &&
             get(/databases/$(database)/documents/system/meta).data.setupDone == true;
    }

    match /system/meta {
      allow read: if true;
      allow create: if isSignedIn() && request.resource.data.setupDone == true;
      allow update: if isSignedIn() && resource.data.setupDone != true
                    && request.resource.data.setupDone == true;
      allow delete: if false;
    }

    match /users/{uid} {
      allow read: if isSignedIn() && (myUid() == uid || isAnyAdmin());
      allow create: if isSignedIn() && (
                      (isAdmin() && (request.resource.data.role == 'employee' || request.resource.data.role == 'assistant_admin')) ||
                      (myUid() == uid && request.resource.data.role == 'admin' && !setupDone())
                    );
      allow update: if isAdmin() || (
                      isSignedIn() && myUid() == uid &&
                      request.resource.data.diff(resource.data).affectedKeys().hasOnly(['name'])
                    );
      allow delete: if isAdmin();
    }

    match /projects/{projectId} {
      allow read: if isAnyAdmin() || (isSignedIn() && myUid() in resource.data.assignees);
      allow create: if isAnyAdmin();
      allow update: if isAnyAdmin() || (
        isSignedIn() && myUid() in resource.data.assignees &&
        request.resource.data.diff(resource.data).affectedKeys().hasOnly(['progressMap'])
      );
      allow delete: if isAnyAdmin();
    }

    match /projectComments/{commentId} {
      allow read: if isAnyAdmin() || (
        isSignedIn() &&
        myUid() in get(/databases/$(database)/documents/projects/$(resource.data.projectId)).data.assignees
      );
      allow create: if isSignedIn() && request.resource.data.uid == myUid() && (
        isAnyAdmin() ||
        myUid() in get(/databases/$(database)/documents/projects/$(request.resource.data.projectId)).data.assignees
      );
      allow update: if false;
      allow delete: if isAnyAdmin() || (isSignedIn() && resource.data.uid == myUid());
    }

    match /chatMessages/{msgId} {
      allow read: if isSignedIn();
      allow create: if isSignedIn() && request.resource.data.uid == myUid()
                    && request.resource.data.text is string
                    && request.resource.data.text.size() > 0
                    && request.resource.data.text.size() <= 2000;
      allow update: if false;
      allow delete: if isAnyAdmin() || (isSignedIn() && resource.data.uid == myUid());
    }

    match /tasks/{taskId} {
      allow read: if isAnyAdmin() || (isSignedIn() && resource.data.assignedTo == myUid());
      allow create: if isAnyAdmin();
      allow update: if isAnyAdmin() || (
        isSignedIn() && resource.data.assignedTo == myUid() &&
        request.resource.data.diff(resource.data).affectedKeys().hasOnly(['status'])
      );
      allow delete: if isAnyAdmin();
    }

    match /achievements/{achId} {
      allow read: if isAnyAdmin() || (isSignedIn() && resource.data.uid == myUid());
      allow create: if isSignedIn() && request.resource.data.uid == myUid();
      allow update, delete: if isAdmin();
    }

    match /requests/{reqId} {
      allow read: if isAnyAdmin() || (isSignedIn() && resource.data.uid == myUid());
      allow create: if isSignedIn() && request.resource.data.uid == myUid()
                    && request.resource.data.status == 'pending';
      allow update: if isAdmin();
      allow delete: if isAdmin();
    }

    match /weeklyReports/{repId} {
      allow read: if isAnyAdmin() || (isSignedIn() && resource.data.uid == myUid());
      allow create: if isSignedIn() && request.resource.data.uid == myUid();
      allow update, delete: if isAdmin();
    }

    match /notifications/{notifId} {
      allow read: if isSignedIn() && resource.data.toUid == myUid();
      allow create: if isSignedIn();
      allow update: if isSignedIn() && resource.data.toUid == myUid()
                    && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['read']);
      allow delete: if isAdmin() || (isSignedIn() && resource.data.toUid == myUid());
    }
  }
}
